pipeline {
    agent any

    stages {
        stage('Clone') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/CSID-DGU/2025-2-OSSPrac-sparkle-01.git'
            }
        }

        stage('Build Image') {
            steps {
                dir('Dockerimages_1/app') {
                    sh 'docker build -t johyerim06/app:latest .'
                }
            }
        }

        stage('DockerHub Login') {
            steps {
                withCredentials([usernamePassword(credentialsId: 'docker-hub', usernameVariable: 'DOCKER_USER', passwordVariable: 'DOCKER_PW')]) {
                    sh '''
                        echo $DOCKER_PW | docker login -u $DOCKER_USER --password-stdin
                    '''
                }
            }
        }

        stage('Push Image') {
            steps {
                sh 'docker push johyerim06/app:latest'
            }
        }
    }
}
